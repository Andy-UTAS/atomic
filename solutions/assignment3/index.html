      # Define a function to return the dispersion

def dispersion_diatomic(k, kappa = 1, m1 = 2, m2 = 1, acoustic=True):

    cons = m1 + m2
    sq = np.sqrt((cons ** 2) - (4 * m1 * m2 * np.sin(k * a/2) ** 2))
    if acoustic:
        sq *= -1
    return np.sqrt(kappa/(m1*m2) * (cons + sq)/m)
